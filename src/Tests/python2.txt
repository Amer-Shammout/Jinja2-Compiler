# تعريف دالة مع استخدام كل أنواع الأعداد والمعرفات
def compute_area(width, height=10, factor=2.5e1):
    # عملية حسابية مع جميع العمليات
    area = (width + height) * factor // 2 - 5 ** 2
    if area >= 100:
        print("Area is large:", area)
    else:
        print('Area is small:', area)
